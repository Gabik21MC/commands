059/**
060 * A Base command is defined as a command group of related commands.
061 * A BaseCommand does not imply nor enforce that they use the same root command.
062 *
063 * It is up to the end user how to organize their command. you could use 1 base command per
064 * command in your application.
065 *
066 * Optionally (and encouraged), you can use the base command to represent a root command, and
067 * then each actionable command is a sub command
068 */

207    /**
208     * This should be called whenever the command gets registered.
209     * It sets all required fields correctly and injects dependencies.
210     *
211     * @param manager
212     *         The manager to register as this command's owner and handler.
213     * @param cmd
214     *         The command name to use register with.
215     */
216    private void onRegister(CommandManager manager, String cmd) {
217        manager.injectDependencies(this);
218        this.manager = manager;
219
220        final Annotations annotations = manager.getAnnotations();
221        final Class<? extends BaseCommand> self = this.getClass();
222
223        String[] cmdAliases = annotations.getAnnotationValues(self, CommandAlias.class, Annotations.REPLACEMENTS | Annotations.LOWERCASE | Annotations.NO_EMPTY);
224
225        if (cmd == null && cmdAliases != null) {
226            cmd = cmdAliases[0];
227        }
228
229        this.commandName = cmd != null ? cmd : self.getSimpleName().toLowerCase();
230        this.permission = annotations.getAnnotationValue(self, CommandPermission.class, Annotations.REPLACEMENTS);
231        this.description = this.commandName + " commands";
232        this.parentSubcommand = getParentSubcommand(self);
233        this.conditions = annotations.getAnnotationValue(self, Conditions.class, Annotations.REPLACEMENTS | Annotations.NO_EMPTY);
234
235        registerSubcommands();
236        registerSubclasses(cmd);
237
238        if (cmdAliases != null) {
239            Set<String> cmdList = new HashSet<>();
240            Collections.addAll(cmdList, cmdAliases);
241            cmdList.remove(cmd);
242            for (String cmdAlias : cmdList) {
243                register(cmdAlias, this);
244            }
245        }
246
247        if (cmd != null) {
248            register(cmd, this);
249        }
250    }

252    /**
253     * This recursively registers all subclasses of the command as subcommands, if they are of type {@link BaseCommand}.
254     *
255     * @param cmd
256     *         The command name of this command.
257     */
258    private void registerSubclasses(String cmd) {
259        for (Class<?> clazz : this.getClass().getDeclaredClasses()) {
260            if (BaseCommand.class.isAssignableFrom(clazz)) {
261                try {
262                    BaseCommand subCommand = null;
263                    Constructor<?>[] declaredConstructors = clazz.getDeclaredConstructors();
264                    for (Constructor<?> declaredConstructor : declaredConstructors) {
265
266                        declaredConstructor.setAccessible(true);
267                        Parameter[] parameters = declaredConstructor.getParameters();
268                        if (parameters.length == 1) {
269                            subCommand = (BaseCommand) declaredConstructor.newInstance(this);
270                        } else {
271                            manager.log(LogLevel.INFO, "Found unusable constructor: " + declaredConstructor.getName() + "(" + Stream.of(parameters).map(p -> p.getType().getSimpleName() + " " + p.getName()).collect(Collectors.joining("<c2>,</c2> ")) + ")");
272                        }
273                    }
274                    if (subCommand != null) {
275                        subCommand.parentCommand = this;
276                        subCommand.onRegister(manager, cmd);
277                        this.subCommands.putAll(subCommand.subCommands);
278                        this.registeredCommands.putAll(subCommand.registeredCommands);
279                    } else {
280                        this.manager.log(LogLevel.ERROR, "Could not find a subcommand ctor for " + clazz.getName());
281                    }
282                } catch (InstantiationException | IllegalAccessException | InvocationTargetException e) {
283                    this.manager.log(LogLevel.ERROR, "Error registering subclass", e);
284                }
285            }
286        }
287    }

289    /**
290     * This registers all subcommands of the command.
291     */
292    private void registerSubcommands() {
293        final Annotations annotations = manager.getAnnotations();
294        boolean foundDefault = false;
295        boolean foundCatchUnknown = false;
296        boolean isParentEmpty = parentSubcommand == null || parentSubcommand.isEmpty();
297
298        for (Method method : this.getClass().getMethods()) {
299            method.setAccessible(true);
300            String sublist = null;
301            String sub = getSubcommandValue(method);
302            final boolean def = annotations.hasAnnotation(method, Default.class);
303            final String helpCommand = annotations.getAnnotationValue(method, HelpCommand.class, Annotations.NOTHING);
304            final String commandAliases = annotations.getAnnotationValue(method, CommandAlias.class, Annotations.NOTHING);
305
306            if (!isParentEmpty && def) {
307                sub = parentSubcommand;
308            }
309            if (isParentEmpty && (def || (!foundDefault && helpCommand != null))) {
310                if (!foundDefault) {
311                    if (def) {
312                        this.subCommands.get(DEFAULT).clear();
313                        foundDefault = true;
314                    }
315                    registerSubcommand(method, DEFAULT);
316                } else {
317                    ACFUtil.sneaky(new IllegalStateException("Multiple @Default/@HelpCommand commands, duplicate on " + method.getDeclaringClass().getName() + "#" + method.getName()));
318                }
319            }
320
321            if (sub != null) {
322                sublist = sub;
323            } else if (commandAliases != null) {
324                sublist = commandAliases;
325            } else if (helpCommand != null) {
326                sublist = helpCommand;
327                hasHelpCommand = true;
328            }
329
330            boolean preCommand = annotations.hasAnnotation(method, PreCommand.class);
331            boolean hasCatchUnknown = annotations.hasAnnotation(method, CatchUnknown.class) ||
332                    annotations.hasAnnotation(method, CatchAll.class) ||
333                    annotations.hasAnnotation(method, UnknownHandler.class);
334
335            if (hasCatchUnknown || (!foundCatchUnknown && helpCommand != null)) {
336                if (!foundCatchUnknown) {
337                    if (hasCatchUnknown) {
338                        this.subCommands.get(CATCHUNKNOWN).clear();
339                        foundCatchUnknown = true;
340                    }
341                    registerSubcommand(method, CATCHUNKNOWN);
342                } else {
343                    ACFUtil.sneaky(new IllegalStateException("Multiple @UnknownHandler/@HelpCommand commands, duplicate on " + method.getDeclaringClass().getName() + "#" + method.getName()));
344                }
345            } else if (preCommand) {
346                if (this.preCommandHandler == null) {
347                    this.preCommandHandler = method;
348                } else {
349                    ACFUtil.sneaky(new IllegalStateException("Multiple @PreCommand commands, duplicate on " + method.getDeclaringClass().getName() + "#" + method.getName()));
350                }
351            }
352            if (Objects.equals(method.getDeclaringClass(), this.getClass()) && sublist != null) {
353                registerSubcommand(method, sublist);
354            }
355        }
356    }

358    /**
359     * Gets the subcommand name of the method given.
360     *
361     * @param method
362     *         The method to check.
363     *
364     * @return The name of the subcommand. It returns null if the input doesn't have {@link Subcommand} attached.
365     */
366    private String getSubcommandValue(Method method) {
367        final String sub = manager.getAnnotations().getAnnotationValue(method, Subcommand.class, Annotations.NOTHING);
368        if (sub == null) {
369            return null;
370        }
371        Class<?> clazz = method.getDeclaringClass();
372        String parent = getParentSubcommand(clazz);
373        return parent == null || parent.isEmpty() ? sub : parent + " " + sub;
374    }
375
376    private String getParentSubcommand(Class<?> clazz) {
377        List<String> subList = new ArrayList<>();
378        while (clazz != null) {
379            String sub = manager.getAnnotations().getAnnotationValue(clazz, Subcommand.class, Annotations.NOTHING);
380            if (sub != null) {
381                subList.add(sub);
382            }
383            clazz = clazz.getEnclosingClass();
384        }
385        Collections.reverse(subList);
386        return ACFUtil.join(subList, " ");
387    }

389    /**
390     * Registers the given {@link BaseCommand cmd} as a child of the {@link RootCommand} linked to the name given.
391     *
392     * @param name
393     *         Name of the parent to cmd.
394     * @param cmd
395     *         The {@link BaseCommand} to add as a child to the {@link RootCommand} owned name field.
396     */
397    private void register(String name, BaseCommand cmd) {
398        String nameLower = name.toLowerCase();
399        RootCommand rootCommand = manager.obtainRootCommand(nameLower);
400        rootCommand.addChild(cmd);
401
402        this.registeredCommands.put(nameLower, rootCommand);
403    }

405    /**
406     * Registers the given {@link Method} as a subcommand.
407     *
408     * @param method
409     *         The method to register as a subcommand.
410     * @param subCommand
411     *         The subcommand's name(s).
412     */
413    private void registerSubcommand(Method method, String subCommand) {
414        subCommand = manager.getCommandReplacements().replace(subCommand.toLowerCase());
415        final String[] subCommandParts = ACFPatterns.SPACE.split(subCommand);
416        // Must run getSubcommandPossibility BEFORE we rewrite it just after this.
417        Set<String> cmdList = getSubCommandPossibilityList(subCommandParts);
418
419        // Strip pipes off for auto complete addition
420        for (int i = 0; i < subCommandParts.length; i++) {
421            String[] split = ACFPatterns.PIPE.split(subCommandParts[i]);
422            if (split.length == 0 || split[0].isEmpty()) {
423                throw new IllegalArgumentException("Invalid @Subcommand configuration for " + method.getName() + " - parts can not start with | or be empty");
424            }
425            subCommandParts[i] = split[0];
426        }
427        String prefSubCommand = ApacheCommonsLangUtil.join(subCommandParts, " ");
428        final String[] aliasNames = manager.getAnnotations().getAnnotationValues(method, CommandAlias.class, Annotations.REPLACEMENTS | Annotations.LOWERCASE);
429
430        String cmdName = aliasNames != null ? aliasNames[0] : this.commandName + " ";
431        RegisteredCommand cmd = manager.createRegisteredCommand(this, cmdName, method, prefSubCommand);
432
433        for (String subcmd : cmdList) {
434            subCommands.put(subcmd, cmd);
435        }
436        cmd.addSubcommands(cmdList);
437
438        if (aliasNames != null) {
439            for (String name : aliasNames) {
440                register(name, new ForwardingCommand(this, subCommandParts));
441            }
442        }
443    }

445    /**
446     * Takes a string like "foo|bar baz|qux" and generates a list of
447     * - foo baz
448     * - foo qux
449     * - bar baz
450     * - bar qux
451     *
452     * For every possible sub command combination
453     *
454     * @param subCommandParts
455     * @return List of all sub command possibilities
456     */
457    private static Set<String> getSubCommandPossibilityList(String[] subCommandParts) {
458        int i = 0;
459        Set<String> current = null;
460        while (true) {
461            Set<String> newList = new HashSet<>();
462
463            if (i < subCommandParts.length) {
464                for (String s1 : ACFPatterns.PIPE.split(subCommandParts[i])) {
465                    if (current != null) {
466                        newList.addAll(current.stream().map(s -> s + " " + s1).collect(Collectors.toList()));
467                    } else {
468                        newList.add(s1);
469                    }
470                }
471            }
472
473            if (i + 1 < subCommandParts.length) {
474                current = newList;
475                i = i + 1;
476                continue;
477            }
478
479            return newList;
480        }
481    }

483    public void execute(CommandIssuer issuer, String commandLabel, String[] args) {
484        commandLabel = commandLabel.toLowerCase();
485        try {
486            CommandOperationContext commandContext = preCommandOperation(issuer, commandLabel, args, false);
487
488            if (args.length > 0) {
489                CommandSearch cmd = findSubCommand(args);
490                if (cmd != null) {
491                    execSubcommand = cmd.getCheckSub();
492                    final String[] execargs = Arrays.copyOfRange(args, cmd.argIndex, args.length);
493                    executeCommand(commandContext, issuer, execargs, cmd.cmd);
494                    return;
495                }
496            }
497
498            if (subCommands.get(DEFAULT) != null && args.length == 0) {
499                findAndExecuteCommand(commandContext, DEFAULT, issuer, args);
500            } else if (subCommands.get(CATCHUNKNOWN) != null) {
501                if (!findAndExecuteCommand(commandContext, CATCHUNKNOWN, issuer, args)) {
502                    help(issuer, args);
503                }
504            } else if (subCommands.get(DEFAULT) != null) {
505                findAndExecuteCommand(commandContext, DEFAULT, issuer, args);
506            }
507
508        } finally {
509            postCommandOperation();
510        }
511    }

538     * This is ran before any command operation has been performed.
539     *
540     * @param issuer
541     *         The user who executed the command.
542     * @param commandLabel
543     *         The label the user used to execute the command. This is not the command name, but their input.
544     *         When there is multiple aliases, this is which alias was used
545     * @param args
546     *         The arguments passed to the command when executing it.
547     * @param isAsync
548     *         Whether the command is executed off of the main thread.
549     *
550     * @return The context which is being registered to the {@link CommandManager}'s {@link
551     * CommandManager#commandOperationContext thread local stack}.
552     */
553    private CommandOperationContext preCommandOperation(CommandIssuer issuer, String commandLabel, String[] args, boolean isAsync) {
554        Stack<CommandOperationContext> contexts = CommandManager.commandOperationContext.get();
555        CommandOperationContext context = this.manager.createCommandOperationContext(this, issuer, commandLabel, args, isAsync);
556        contexts.push(context);
557        lastCommandOperationContext = context;
558        execSubcommand = null;
559        execLabel = commandLabel;
560        origArgs = args;
561        return context;
562    }

596    /**
597     * Finds a subcommand of the given arguments.
598     *
599     * @param args
600     *         The arguments the user input.
601     * @param completion
602     *         Whether or not completion of arguments should kick in. This may end up with worse than wanted results.
603     *
604     * @return The identified subcommand.
605     */
606    private CommandSearch findSubCommand(String[] args, boolean completion) {
607        for (int i = args.length; i >= 0; i--) {
608            String checkSub = ApacheCommonsLangUtil.join(args, " ", 0, i).toLowerCase();
609            Set<RegisteredCommand> cmds = subCommands.get(checkSub);
610
611            final int extraArgs = args.length - i;
612            if (!cmds.isEmpty()) {
613                RegisteredCommand cmd = null;
614                if (cmds.size() == 1) {
615                    cmd = ACFUtil.getFirstElement(cmds);
616                } else {
617                    Optional<RegisteredCommand> optCmd = cmds.stream().filter(c -> {
618                        int required = c.requiredResolvers;
619                        int optional = c.optionalResolvers;
620                        return extraArgs <= required + optional && (completion || extraArgs >= required);
621                    }).min((c1, c2) -> {
622                        int a = c1.consumeInputResolvers;
623                        int b = c2.consumeInputResolvers;
624
625                        if (a == b) {
626                            return 0;
627                        }
628                        return a < b ? 1 : -1;
629                    });
630                    if (optCmd.isPresent()) {
631                        cmd = optCmd.get();
632                    }
633                }
634                if (cmd != null) {
635                    return new CommandSearch(cmd, i, checkSub);
636                }
637            }
638        }
639        return null;
640    }

642    private void executeCommand(CommandOperationContext commandOperationContext,
643                                CommandIssuer issuer, String[] args, RegisteredCommand cmd) {
644        if (cmd.hasPermission(issuer)) {
645            commandOperationContext.setRegisteredCommand(cmd);
646            if (checkPrecommand(commandOperationContext, cmd, issuer, args)) {
647                return;
648            }
649            List<String> sargs = Arrays.asList(args);
650            cmd.invoke(issuer, sargs, commandOperationContext);
651        } else {
652            issuer.sendMessage(MessageType.ERROR, MessageKeys.PERMISSION_DENIED);
653        }
654    }

699     */
700    @SuppressWarnings("WeakerAccess")
701    public List<String> tabComplete(CommandIssuer issuer, String commandLabel, String[] args, boolean isAsync)
702        throws IllegalArgumentException {
703
704        commandLabel = commandLabel.toLowerCase();
705        if (args.length == 0) {
706            args = new String[]{""};
707        }
708        try {
709            CommandOperationContext commandOperationContext = preCommandOperation(issuer, commandLabel, args, isAsync);
710
711            final CommandSearch search = findSubCommand(args, true);
712
713
714            final List<String> cmds = new ArrayList<>();
715
716            if (search != null) {
717                cmds.addAll(completeCommand(issuer, search.cmd, Arrays.copyOfRange(args, search.argIndex, args.length), commandLabel, isAsync));
718            } else if (subCommands.get(CATCHUNKNOWN).size() == 1) {
719                cmds.addAll(completeCommand(issuer, ACFUtil.getFirstElement(subCommands.get(CATCHUNKNOWN)), args, commandLabel, isAsync));
720            } else if (subCommands.get(DEFAULT).size() == 1) {
721                cmds.addAll(completeCommand(issuer, ACFUtil.getFirstElement(subCommands.get(DEFAULT)), args, commandLabel, isAsync));
722            }
723
724            return filterTabComplete(args[args.length - 1], cmds);
725        } finally {
726            postCommandOperation();
727        }
728    }

730    /**
731     * Gets all subcommands which are possible to tabcomplete.
732     *
733     * @param issuer
734     *         The command issuer.
735     * @param args
736     *
737     * @return
738     */
739    List<String> getCommandsForCompletion(CommandIssuer issuer, String[] args) {
740        final Set<String> cmds = new HashSet<>();
741        final int cmdIndex = Math.max(0, args.length - 1);
742        String argString = ApacheCommonsLangUtil.join(args, " ").toLowerCase();
743        for (Map.Entry<String, RegisteredCommand> entry : subCommands.entries()) {
744            final String key = entry.getKey();
745            if (key.startsWith(argString) && !CATCHUNKNOWN.equals(key) && !DEFAULT.equals(key)) {
746                final RegisteredCommand value = entry.getValue();
747                if (!value.hasPermission(issuer) || value.isPrivate) {
748                    continue;
749                }
750
751                String[] split = ACFPatterns.SPACE.split(value.prefSubCommand);
752                cmds.add(split[cmdIndex]);
753            }
754        }
755        return new ArrayList<>(cmds);
756    }

758    /**
759     * Complete a command properly per issuer and input.
760     *
761     * @param issuer
762     *         The user who executed this.
763     * @param cmd
764     *         The command to be completed.
765     * @param args
766     *         All arguments given by the user.
767     * @param commandLabel
768     *         The command name the user used.
769     * @param isAsync
770     *         Whether the command was executed async.
771     *
772     * @return All results to complete the command.
773     */
774    private List<String> completeCommand(CommandIssuer issuer, RegisteredCommand cmd, String[] args, String commandLabel, boolean isAsync) {
775        if (!cmd.hasPermission(issuer) || args.length > cmd.consumeInputResolvers || args.length == 0 || cmd.complete == null) {
776            return Collections.emptyList();
777        }
778
779        List<String> cmds = manager.getCommandCompletions().of(cmd, issuer, args, isAsync);
780        return filterTabComplete(args[args.length-1], cmds);
781    }

783    /**
784     * Gets the actual args in string form the user typed
785     * This returns a list of all tab complete options which are possible with the given argument and commands.
786     * @param arg
787     *         Argument which was pressed tab on.
788     * @param cmds
789     *         The possibilities to return.
790     *
791     * @return All possible options. This may be empty.
792     */
793    private static List<String> filterTabComplete(String arg, List<String> cmds) {
794        return cmds.stream()
795                   .distinct()
796                   .filter(cmd -> cmd != null && (arg.isEmpty() || ApacheCommonsLangUtil.startsWithIgnoreCase(cmd, arg)))
797                   .collect(Collectors.toList());
798    }

813     * Gets a registered command under the given name.
814     * If requireOne is true, it won't accept more than a single matching subcommand.
815     *
816     * @param subcommand
817     *         Name of the subcommand wanted.
818     * @param requireOne
819     *         Whether to only accept 1 result.
820     *
821     * @return The subcommand found, or null if none/too many.
822     */
823    private RegisteredCommand getCommandBySubcommand(String subcommand, boolean requireOne) {
824        final Set<RegisteredCommand> commands = subCommands.get(subcommand);
825        if (!commands.isEmpty() && (!requireOne || commands.size() == 1)) {
826            return commands.iterator().next();
827        }
828        return null;

831    /**
832     * Internally calls {@link #executeCommand(CommandOperationContext, CommandIssuer, String[], RegisteredCommand)}
833     * and gets through {@link #getCommandBySubcommand(String)}.
834     *
835     * @param commandContext
836     *         The command context to use.
837     * @param subcommand
838     *         The subcommand to find the executor of.
839     * @param issuer
840     *         The issuer who executed the subcommand.
841     * @param args
842     *         All arguments given by the issuer.
843     *
844     * @return Whether it found a command or not.
845     *
846     * @see #executeCommand(CommandOperationContext, CommandIssuer, String[], RegisteredCommand)
847     * @see #getCommandBySubcommand(String)
848     * @see RegisteredCommand#invoke(CommandIssuer, List, CommandOperationContext)
849     */
850    private boolean findAndExecuteCommand(CommandOperationContext commandContext, String subcommand, CommandIssuer issuer, String... args) {
851        final RegisteredCommand cmd = this.getCommandBySubcommand(subcommand);
852        if (cmd != null) {
853            executeCommand(commandContext, issuer, args, cmd);
854            return true;
855        }
856
857        return false;
858    }

860    /**
861     * Executes the precommand and sees whether something is wrong. Ideally, you get false from this.
862     *
863     * @param commandOperationContext
864     *         The context to use.
865     * @param cmd
866     *         The command executed.
867     * @param issuer
868     *         The issuer who executed the command.
869     * @param args
870     *         The arguments the issuer provided.
871     *
872     * @return Whether something went wrong.
873     */
874    private boolean checkPrecommand(CommandOperationContext commandOperationContext, RegisteredCommand cmd, CommandIssuer issuer, String[] args) {
875        Method pre = this.preCommandHandler;
876        if (pre != null) {
877            try {
878                Class<?>[] types = pre.getParameterTypes();
879                Object[] parameters = new Object[pre.getParameterCount()];
880                for (int i = 0; i < parameters.length; i++) {
881                    Class<?> type = types[i];
882                    Object issuerObject = issuer.getIssuer();
883                    if (manager.isCommandIssuer(type) && type.isAssignableFrom(issuerObject.getClass())) {
884                        parameters[i] = issuerObject;
885                    } else if (CommandIssuer.class.isAssignableFrom(type)) {
886                        parameters[i] = issuer;
887                    } else if (RegisteredCommand.class.isAssignableFrom(type)) {
888                        parameters[i] = cmd;
889                    } else if (String[].class.isAssignableFrom((type))) {
890                        parameters[i] = args;
891                    } else {
892                        parameters[i] = null;
893                    }
894                }
895
896                return (boolean) pre.invoke(this, parameters);
897            } catch (IllegalAccessException | InvocationTargetException e) {
898                this.manager.log(LogLevel.ERROR, "Exception encountered while command pre-processing", e);
899            }
900        }
901        return false;
902    }

981        CommandSearch(RegisteredCommand cmd, int argIndex, String checkSub) {

984            this.checkSub = checkSub;

991        @Override

993            if (this == o) return true;
994            if (o == null || getClass() != o.getClass()) return false;
995            CommandSearch that = (CommandSearch) o;
996            return argIndex == that.argIndex &&
997                    Objects.equals(cmd, that.cmd) &&
998                    Objects.equals(checkSub, that.checkSub);
999        }

1004        }
